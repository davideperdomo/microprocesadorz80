# Cargar valores B, C
# contador
LD B,00H
# limite num - 2
LD C,07H
# reg aux
LD D,00H
LD E,01H
LD H,00H
# while (B < C) { H = D + E; OUT H; D = E; E = H; }

# body del while { d = d + 2; out d }
LD A,D

LD D,E
LD E,H
# Update
LD A,B

ENDWHILE LD A,H
